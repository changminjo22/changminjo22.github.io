Linked List를 이용해 Queue, Stack을 구현하고, 나만의 정렬 알고리즘을 만들어 보았다.

1. Queue의 구현

Queue를 Linked List로 구현해 보았다.

```c
#include<stdio.h>
#include<malloc.h>

typedef struct node{
    int data;
    struct node *next;
} Node;

void printAllElements(Node *front) {
    printf("All Elements are: ");
    for ( ; front != NULL; front = front -> next) {
        printf("%d ", front -> data);
    }
    printf("\n");
}

Node *getElement(Node *front, int i) {
    for (int k=0; k<i && front; k++) {
        front = front -> next;
    }
    return front;
}

void insertElement(Node *prev, Node *novel) {
    novel -> next = prev -> next;
    prev -> next = novel;
}

void deleteElement(Node *prev) {
    Node *curr = prev -> next;
    prev -> next = curr -> next;
    free(curr);
}

Node *front, *rear;

void newQueue(){
    front = NULL;
    rear = NULL;
}

void enQueue(int x) {
    Node *ptr;
    ptr = (Node *)malloc(sizeof(Node));
    ptr -> data = x;
    ptr -> next = NULL;
    if (front == NULL) front = ptr;
    else rear -> next = ptr;
    rear = ptr;
}

int deQueue() {
    int item;
    Node *tmp = front;

    if (tmp == NULL) {
        return -1;
    }

    item = tmp -> data;
    front = tmp -> next;
    free(tmp);
    if (front == NULL) rear = front;
    return item;
}

int main() {
    newQueue();
    int Q; scanf("%d", &Q);
    while (Q--) {
        char str[10]; scanf("%s", str);
        if (str[0] == 'p') {
            if (str[1] == 'u') { // push
                int tmp; scanf("%d", &tmp);
                enQueue(tmp);
            }
            else { // pop
                printf("%d\n", deQueue());
            }
        }
        else if (str[0] == 'f') {
            if (front != NULL) printf("%d\n", front -> data);
            else printf("-1\n");
        }
        else if (str[0] == 'b') {
            if (rear != NULL) printf("%d\n", rear -> data);
            else printf("-1\n");
        }
        else if (str[0] == 'e') {
            if (front == NULL) printf("1\n");
            else printf("0\n");
        }
        else { // stack size
            Node *i = front;
            if (i == NULL) {
                printf("0\n");
            }
            else {
                int num = 0;
                for ( ; i != NULL; i = i -> next) {
                    num = num + 1;
                }
                printf("%d\n", num);
            }
        }
        // printAllElements(front);
    }
}
```

2. Stack의 구현

Stack을 Linked List로 구현해 보았다.

```c
#include<stdio.h>
#include<malloc.h>

typedef struct node{
    int data;
    struct node *next;
} Node;

void printAllElements(Node *front) {
    for ( ; front != nullptr; front = front -> next) {
        printf("%d ", front -> data);
    }
    printf("\n");
}

Node *getElement(Node *front, int i) {
    for (int k=0; k<i && front; k++) {
        front = front -> next;
    }
    return front;
}

void insertElement(Node *prev, Node *novel) {
    novel -> next = prev -> next;
    prev -> next = novel;
}

void deleteElement(Node *prev) {
    Node *curr = prev -> next;
    prev -> next = curr -> next;
    free(curr);
}

Node *top;

void push(int x) {
    Node *ptr;
    ptr = (Node *)malloc(sizeof(Node));
    ptr -> data = x;
    ptr -> next = top;
    top = ptr;
}

int pop() {
    int item;
    Node *tmp = top;

    if (tmp == nullptr) {
        return -1;
    }
    else {
        item = tmp -> data;
        top = tmp -> next;
        free(tmp);
        return item;
    }
}

void newStack() {
    top = nullptr;
}

int main() {
    int Q; scanf("%d", &Q);
    while (Q--) {
        char str[10]; scanf("%s", str);
        if (str[0] == 'p') {
            if (str[1] == 'u') { // push
                int tmp; scanf("%d", &tmp);
                push(tmp);
            }
            else { // pop
                printf("%d\n", pop());
            }
        }
        else if (str[0] == 't') {
            if (top != nullptr) printf("%d\n", top -> data);
            else printf("-1\n");
        }
        else if (str[0] == 'e') {
            if (top == nullptr) printf("1\n");
            else printf("0\n");
        }
        else { // stack size
            Node *i = top;
            if (i == nullptr) {
                printf("0\n");
            }
            else {
                int num = 0;
                for ( ; i != nullptr; i = i -> next) {
                    num = num + 1;
                }
                printf("%d\n", num);
            }
        }
    }
}
```

3. $O(N\sqrt{N})$ 정렬 알고리즘

직접 만들었는데 이미 알려져 있는진 잘 모르겠다.

```c
#include<bits/stdc++.h>

typedef struct dnode {
    int data;
    struct dnode *prev;
    struct dnode *next;
} DNode;

int N, d[101010];
int bucket_sz = 1;
DNode *first;
DNode *bucket[101010];

void insertNodePrev(DNode *prev, DNode *ptr) {
    ptr->prev = prev;
    if (prev->next == nullptr) {
        ptr->next = nullptr;
        prev->next = ptr;
    }
    else {
        ptr->next = prev->next;
        prev->next->prev = ptr;
        prev->next = ptr;
    }
}

void insertNodeNext(DNode *next, DNode *ptr) {
    ptr->next = next;
    if (next->prev == nullptr) {
        ptr->prev = nullptr;
        next->prev = ptr;
    }
    else {
        ptr->prev = next->prev;
        next->prev->next = ptr;
        next->prev = ptr;
    }
}

void swap_node(DNode *A, DNode *B) {
    if (A == NULL || B->next == NULL) return;

    // Swapping A and B
    DNode* temp;
    temp = A->next;
    A->next = B->next;
    B->next = temp;

    if (A->next != NULL) A->next->prev = A;
    if (B->next != NULL) B->next->prev = B;

    temp = A->prev;
    A->prev = B->prev;
    B->prev = temp;

    if (A->prev != NULL) A->prev->next = A;
    if (B->prev != NULL) B->prev->next = B;
}

void swap_data(DNode *A, DNode *B) {
    int tmp; tmp = A->data;
    A->data = B->data, B->data = tmp;
}

void deleteNode(DNode *ptr) {
    ptr->prev->next = ptr->next;
    ptr->next->prev = ptr->prev;
    free(ptr);
}

DNode *getElement(DNode *front, int i) {
    for (int k = 0; k < i && front; k++) {
        front = front->next;
    }
    return front;
}

void printAllElements(DNode *front) {
    for (; front != nullptr; front = front->next) {
        printf("%d\n", front->data);
    }
    // printf("\n");
}

void DetermineFirst() {
    while (first -> prev != nullptr) first = first -> prev;
}

void interval_bubble(DNode *s, DNode *ptr) {
    while (s->data <= ptr->data && s->next != nullptr) s = s->next;
    if (s->next == nullptr) {
        if (s->data <= ptr->data) insertNodePrev(s, ptr);
        else insertNodeNext(s, ptr);
    }
    else insertNodeNext(s, ptr);
}

void rearrange_idx(int idx) {
    for (int i=idx; i<=bucket_sz; i++) {
        bucket[i] = bucket[i]->prev;
    }
}

void online_sort(DNode *ptr) {
    if (ptr->data >= bucket[bucket_sz]->data) {
        interval_bubble(bucket[bucket_sz], ptr);
    }
    else if (ptr->data <= bucket[1]->data) {
        insertNodeNext(bucket[1], ptr);
        bucket[1] = bucket[1]->prev;
        rearrange_idx(2);
    }
    else {
        for (int i=1; i<=bucket_sz; i++) {
            if (ptr->data <= bucket[i]->data) {
                interval_bubble(bucket[i-1], ptr);
                rearrange_idx(i); break;
            }
        }
    }
}

void bucket_add() {
    DNode *seq = bucket[bucket_sz];
    for (int i=1; i<=sqrt(N); i++) seq = seq -> next;
    bucket[++bucket_sz] = seq;
}

int main() {
    scanf("%d", &N);
    for (int i = 1; i <= N; i++) scanf("%d", &d[i]);
    if (N == 1) {
        printf("%d", d[1]);
        return 0;
    }

    first = (DNode *) malloc(sizeof(DNode));
    first->data = d[1], first->next = nullptr, first->prev = nullptr;
    bucket[1] = first;
    for (int i = 2; i <= N; i++) {
        DNode *cur;
        cur = (DNode *) malloc(sizeof(DNode));
        cur->data = d[i];
        cur->prev = nullptr, cur->next = nullptr;
        online_sort(cur);
        if (i % int(sqrt(N)) == 1) bucket_add();
        // DetermineFirst();
        // printAllElements(first);
    }
    DetermineFirst();
    printAllElements(first);

    return 0;
}
```
